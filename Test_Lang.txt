int Gallician_War (int Caesar)
{
	int galli = 0;
	if (Caesar <= galli)
		return 1;
	return Caesar * Gallician_War (Caesar - 1);
}

int main()
{
	int Caesar = 0;
	scanf ("%d", &Caesar);
	printf ("%d", Caesar);
	int gonets = Gallician_War (Caesar);
	printf ("\nИтоги войны равны ");
	printf ("%d", gonets);
	return 0;
}
